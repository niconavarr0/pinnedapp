/**
 * Pinned SDK v0.1.0
 * Cross-platform calendar scheduling widget for chat applications
 * License: Commercial — contact [email protected]
 */

(function(global) {
  const Pinned = {

    /** Attach the Pinned widget to a chat message container */
    attach: function({ container, participants, onConfirm, theme = 'light' }) {
      const el = typeof container === 'string'
        ? document.querySelector(container)
        : container;

      if (!el) throw new Error('[Pinned] Container not found: ' + container);

      const widget = new PinnedWidget({ participants, onConfirm, theme });
      el.appendChild(widget.render());
      return widget;
    },

    /** Generate a calendar invite file (.ics) for an event */
    exportToCalendar: function(event) {
      return CalendarExport.generateICS(event);
    },

    version: '0.1.0'
  };

  global.Pinned = Pinned;
})(typeof window !== 'undefined' ? window : this);
/**
 * PinnedWidget — renders the calendar UI inside a chat bubble
 */
class PinnedWidget {
  constructor({ participants, onConfirm, theme }) {
    this.participants = participants;
    this.onConfirm    = onConfirm;
    this.theme        = theme;
    this.selectedDate = null;
    this.selectedTime = '12:00';
    this.month        = new Date().getMonth();
    this.year         = new Date().getFullYear();
  }

  render() {
    const wrapper = document.createElement('div');
    wrapper.className = 'pinned-widget-root';
    wrapper.setAttribute('data-theme', this.theme);
    wrapper.innerHTML = this._buildHTML();
    this._bindEvents(wrapper);
    return wrapper;
  }

  _buildHTML() {
    const days = this._getDaysInMonth(this.year, this.month);
    return `
      <div class="pinned-header">
        <span class="pinned-icon">📌</span>
        <span class="pinned-title">Pick a date</span>
      </div>
      <div class="pinned-calendar">
        ${days.map(d => `<button class="pinned-day" data-date="${d}">${d}</button>`).join('')}
      </div>
      <button class="pinned-confirm">Confirm date</button>
    `;
  }

  _bindEvents(root) {
    root.querySelectorAll('.pinned-day').forEach(btn => {
      btn.addEventListener('click', () => {
        this.selectedDate = btn.dataset.date;
        this.onConfirm?.({
          date: this.selectedDate,
          time: this.selectedTime,
          participants: this.participants
        });
      });
    });
  }

  _getDaysInMonth(year, month) {
    return new Array(new Date(year, month + 1, 0).getDate())
      .fill(0).map((_, i) => i + 1);
  }
}
/**
 * CalendarExport — generates .ics files for Apple, Google, Outlook
 */
const CalendarExport = {

  generateICS: function({ date, time, title, participants }) {
    const start  = this._toICSDate(date, time);
    const end    = this._toICSDate(date, time, 60); // +60 min
    const uid    = crypto.randomUUID();
    const stamp  = new Date().toISOString().replace(/[-:]/g, '').split('.')[0] + 'Z';

    const attendees = participants
      .map(p => `ATTENDEE:mailto:${p}@example.com`)
      .join('\n');

    const ics = [
      'BEGIN:VCALENDAR',
      'VERSION:2.0',
      'PRODID:-//Pinned//SDK//EN',
      'BEGIN:VEVENT',
      `UID:${uid}`,
      `DTSTAMP:${stamp}`,
      `DTSTART:${start}`,
      `DTEND:${end}`,
      `SUMMARY:${title || 'Pinned event'}`,
      attendees,
      'END:VEVENT',
      'END:VCALENDAR'
    ].join('\n');

    return this._download(ics, 'pinned-event.ics');
  },

  _toICSDate: function(dateStr, timeStr, offsetMin = 0) {
    const d = new Date(`${dateStr}T${timeStr}`);
    d.setMinutes(d.getMinutes() + offsetMin);
    return d.toISOString().replace(/[-:]/g, '').split('.')[0] + 'Z';
  },

  _download: function(content, filename) {
    const blob = new Blob([content], { type: 'text/calendar' });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url; a.download = filename; a.click();
    URL.revokeObjectURL(url);
  }
};
How platforms use it
Step 01
Platform loads SDK
One script tag. No build step required. Works alongside any existing chat architecture.
Step 02
Widget attaches to message
When a user taps 📌, Pinned renders a calendar widget inside the chat bubble. Both users see it live.
Step 03
Both calendars updated
On confirmation, Pinned fires onConfirm with the event data. Platform saves to both users' calendars instantly.
